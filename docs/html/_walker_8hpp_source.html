138  void walkerThreadWork(int tid)
139  {
140  std::string dir;
141  DIR *dirstruct;
142  struct dirent *ent;
143  std::string separ;
144  std::string fullpath;
145 
146  while (true)
147  {
148  while (!m_Halted && !m_DirectoryQueue.try_pop(dir))
149  {
150  // Busy-wait
151  std::this_thread::yield();
152  }
153 
154  // Exit flag
155  if (m_Halted)
156  {
157  break;
158  }
159 
160  separ = (*dir.rbegin() != '/' ? "/" : ""); // *dir.rbegin() is equivalent to dir.back() of c++11
161 
162  // Traverse the directory
163  if ((dirstruct = opendir(dir.c_str())) != NULL)
164  {
165  while ((ent = readdir(dirstruct)) != NULL)
166  {
167  std::stringstream ss;
168 
169  // ignore . and ..
170  if(strcmp(ent->d_name, ".") == 0 || strcmp(ent->d_name, "..") == 0)
171  {
172  continue;
173  }
174 
175  ss << dir << separ << ent->d_name;
176  fullpath = ss.str();
177 
178  // Hash and check if the file is to be ignored.
179  // This has a probability of 1/size_t of false-positives
180  if(m_SkipListHashes.count(m_HashFunction(fullpath)) != 0)
181  {
182  continue;
183  }
184 
185  // Push dirs to walker threads
186  if (ent->d_type == DT_DIR)
187  {
188  m_DirectoryQueue.emplace(fullpath);
189  }
190 
191  // Push path to stat threads
192  mystat(fullpath);
193  }
194 
195  closedir(dirstruct);
196  }
197  else
198  {
199  std::cerr << "-- Error stating directory: " << dir << "\n";
200  }
201  }

230  // Convert all skipped paths to hashes - performance baby
231  for(const std::string& p : skipList)
232  {

237  mystat(path);
238

240  m_DirectoryQueue.push(path);
241 
242  // Start walker threads

246  }
247
